140static u64 Hash(const char* _fileName, u32 _lineNumber)
141{
142 u64 res = _lineNumber;
143 for (const char* At = _fileName; *At; ++At)
144 {
145 res = res * 65599 + *At;
146 }
147 res = ((res << 16) ^ (res >> 16)) * 73244475;
148 res = ((res << 16) ^ (res >> 16)) * 73244475;
149 res = ((res << 16) ^ (res >> 16));
150 return res;
151}
