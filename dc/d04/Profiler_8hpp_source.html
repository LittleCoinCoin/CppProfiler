131static u64 Hash(const char* _fileName, u32 _lineNumber)
132{
133 u64 res = _lineNumber;
134 for (const char* At = _fileName; *At; ++At)
135 {
136 res = res * 65599 + *At;
137 }
138 res = ((res << 16) ^ (res >> 16)) * 73244475;
139 res = ((res << 16) ^ (res >> 16)) * 73244475;
140 res = ((res << 16) ^ (res >> 16));
141 return res;
142}
