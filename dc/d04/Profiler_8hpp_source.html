117static u64 Hash(const char* _fileName, u32 _lineNumber)
118{
119 u64 res = _lineNumber;
120 for (const char* At = _fileName; *At; ++At)
121 {
122 res = res * 65599 + *At;
123 }
124 res = ((res << 16) ^ (res >> 16)) * 73244475;
125 res = ((res << 16) ^ (res >> 16)) * 73244475;
126 res = ((res << 16) ^ (res >> 16));
127 return res;
128}
