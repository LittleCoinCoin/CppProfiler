129static u64 Hash(const char* _fileName, u32 _lineNumber)
130{
131 u64 res = _lineNumber;
132 for (const char* At = _fileName; *At; ++At)
133 {
134 res = res * 65599 + *At;
135 }
136 res = ((res << 16) ^ (res >> 16)) * 73244475;
137 res = ((res << 16) ^ (res >> 16)) * 73244475;
138 res = ((res << 16) ^ (res >> 16));
139 return res;
140}
